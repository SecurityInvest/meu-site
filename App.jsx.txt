// ============================================================
// Calculadora Financeira
// ============================================================

function Calculadora() {
  const [meta, setMeta] = useState("");
  const [preco, setPreco] = useState("97.90");
  const [ads, setAds] = useState("70");
  const [dias, setDias] = useState("30");
  const [resultado, setResultado] = useState(null);

  const calcular = () => {
    const m = parseFloat(meta),
          p = parseFloat(preco),
          a = parseFloat(ads),
          d = parseInt(dias);

    if (isNaN(m) || isNaN(p) || isNaN(a) || isNaN(d)) return;

    const liquido = p - (p * 0.0499 + 1);
    const totalAds = a * d;
    const vendasNecessarias = Math.ceil((m + totalAds) / liquido);
    const cliquesNecessarios = Math.ceil(vendasNecessarias / 0.02);

    setResultado({
      vendasNecessarias,
      vendasDia: (vendasNecessarias / d).toFixed(1),
      totalAds: totalAds.toFixed(2),
      liquido: liquido.toFixed(2),
      breakeven: Math.ceil(totalAds / liquido),
      cliquesNecessarios,
      cpcEstimado: (totalAds / cliquesNecessarios).toFixed(2),
      faturamentoBruto: (vendasNecessarias * p).toFixed(2),
    });
  };

  const campos = [
    { label: "Meta de lucro (R$)", val: meta, set: setMeta, ph: "ex: 3000" },
    { label: "Preço do produto (R$)", val: preco, set: setPreco, ph: "ex: 97.90" },
    { label: "Orçamento/dia (R$)", val: ads, set: setAds, ph: "ex: 70" },
    { label: "Dias de campanha", val: dias, set: setDias, ph: "ex: 30" }
  ];

  return (
    <div className="space-y-4">
      <div className="bg-gray-900 border border-gray-700 rounded-xl p-4 text-center">
        <p className="text-white font-bold">💰 Calculadora de Meta Financeira</p>
        <p className="text-gray-400 text-xs mt-1">Descobre exatamente o que você precisa para atingir sua meta</p>
      </div>

      <div className="grid grid-cols-2 gap-3">
        {campos.map((f, i) => (
          <div key={i}>
            <label className="text-gray-400 text-xs mb-1 block">{f.label}</label>
            <input
              value={f.val}
              onChange={e => f.set(e.target.value)}
              placeholder={f.ph}
              className="w-full bg-gray-800 border border-gray-700 rounded-lg px-3 py-2 text-white text-sm outline-none focus:border-orange-500"
            />
          </div>
        ))}
      </div>

      <button
        onClick={calcular}
        className="w-full bg-gradient-to-r from-orange-500 to-amber-500 text-white font-bold py-3 rounded-xl text-sm"
      >
        ⚡ Calcular Agora
      </button>

      {resultado && (
        <div className="space-y-2">
          <div className="bg-gradient-to-r from-green-900 to-emerald-900 border border-green-700 rounded-xl p-4">
            <p className="text-green-300 text-xs mb-1">Para atingir sua meta você precisa de:</p>
            <p className="text-white font-bold text-2xl">{resultado.vendasNecessarias} vendas</p>
            <p className="text-green-300 text-xs">({resultado.vendasDia} vendas por dia)</p>
          </div>
          {[
            ["💵 Líquido por venda (Kiwify)", `R$${resultado.liquido}`],
            ["📢 Total investido em ads", `R$${resultado.totalAds}`],
            ["⚖️ Breakeven", `${resultado.breakeven} vendas`],
            ["👆 Cliques necessários (2%)", `${resultado.cliquesNecessarios.toLocaleString()}`],
            ["💲 CPC máximo suportado", `R$${resultado.cpcEstimado}`],
            ["📈 Faturamento bruto", `R$${parseFloat(resultado.faturamentoBruto).toLocaleString("pt-BR", {minimumFractionDigits:2})}`],
          ].map(([label, val], i) => (
            <div key={i} className="flex justify-between bg-gray-800 rounded-lg px-3 py-2">
              <span className="text-gray-400 text-xs">{label}</span>
              <span className="text-white font-bold text-xs">{val}</span>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}

// ============================================================
// Gerador de Copy
// ============================================================

function GeradorCopy() {
  const [tipo, setTipo] = useState("anuncio");
  const [nicho, setNicho] = useState("");
  const [preco, setPreco] = useState("");
  const [publico, setPublico] = useState("");
  const [resultado, setResultado] = useState("");
  const [loading, setLoading] = useState(false);
  const [copied, setCopied] = useState(false);

  const gerar = async () => {
    if (!nicho.trim()) return;
    setLoading(true); setResultado("");

    const prompts = {
      anuncio: `Crie um anúncio para Meta Ads para um produto digital sobre "${nicho}" voltado para "${publico||"iniciantes"}" por R$${preco||"97,90"}. Use a fórmula DAS. Máximo 150 palavras. Responda apenas com o texto do anúncio.`
    };

    try {
      const response = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Authorization": "Bearer SUA_API_KEY_AQUI"
        },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 500,
          messages: [{ role: "user", content: prompts[tipo] }]
        })
      });
      const data = await response.json();
      setResultado(data.completion || "Erro ao gerar.");
    } catch {
      setResultado("Erro de conexão.");
    }

    setLoading(false);
  };

  const copiar = () => {
    navigator.clipboard?.writeText(resultado);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  return (
    <div className="space-y-3">
      <div className="bg-gray-900 border border-gray-700 rounded-xl p-4 text-center">
        <p className="text-white font-bold">✍️ Gerador de Copy com IA</p>
        <p className="text-gray-400 text-xs mt-1">Cria anúncios, headlines e e-mails em segundos</p>
      </div>

      <div className="flex gap-2 overflow-x-auto pb-1">
        {["anuncio"].map(v => (
          <button
            key={v}
            onClick={() => setTipo(v)}
            className={`flex-shrink-0 px-3 py-1.5 rounded-lg text-xs font-bold ${tipo===v?"bg-purple-600 text-white":"bg-gray-800 text-gray-400"}`}
          >
            📢 Anúncio
          </button>
        ))}
      </div>

      <div className="space-y-2">
        <input
          value={nicho}
          onChange={e => setNicho(e.target.value)}
          placeholder="Tema do produto (ex: vender online com IA)"
          className="w-full bg-gray-800 border border-gray-700 rounded-lg px-3 py-2 text-white text-sm outline-none focus:border-orange-500"
        />
        <input
          value={publico}
          onChange={e => setPublico(e.target.value)}
          placeholder="Público-alvo (ex: iniciantes)"
          className="w-full bg-gray-800 border border-gray-700 rounded-lg px-3 py-2 text-white text-sm outline-none focus:border-orange-500"
        />
        <input
          value={preco}
          onChange={e => setPreco(e.target.value)}
          placeholder="Preço (R$)"
          className="w-full bg-gray-800 border border-gray-700 rounded-lg px-3 py-2 text-white text-sm outline-none focus:border-orange-500"
        />
      </div>

      <button onClick={gerar} className="w-full bg-gradient-to-r from-purple-500 to-pink-500 text-white font-bold py-3 rounded-xl text-sm">
        {loading ? "Gerando..." : "⚡ Gerar Copy"}
      </button>

      {resultado && (
        <div className="bg-gray-800 p-3 rounded-lg relative">
          <pre className="text-white text-xs whitespace-pre-wrap">{resultado}</pre>
          <button onClick={copiar} className="absolute top-2 right-2 text-xs text-green-400">
            {copied ? "✅ Copiado!" : "📋 Copiar"}
          </button>
        </div>
      )}
    </div>
  );
}

// ============================================================
// App Principal

export default function App() {
  return (
    <div className="p-4 space-y-6 bg-gray-900 min-h-screen">
      <Calculadora />
      <GeradorCopy />
    </div>
  );
}
